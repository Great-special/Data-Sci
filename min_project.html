# Utilize np.select for efficient condition-based assignment
one_d_array = daily_returns.values.flatten()

conditions = [
    (one_d_array > 0),
    (one_d_array < 0),
    (one_d_array == 0)
]

# Specify corresponding choices for each condition
choices = ['Up', 'Down', 'No Change']

# Employ np.select to efficiently apply conditions and assign values to 'Trend' column
flatten_trend = np.select(conditions, choices, default='No Change')

# Assign the reshaped 'Trend' values back to the DataFrame
stocks_data[('Trend', 'AAPL')] = flatten_trend[:1006]
stocks_data[('Trend', 'AMZN')] = flatten_trend[1006:2*1006]
stocks_data[('Trend', 'GOOGL')] = flatten_trend[2*1006:3*1006]
stocks_data[('Trend', 'MSFT')] = flatten_trend[3*1006:]

8. Visualize trend frequency through a Pie Chart:
In[]:
tickers = ['AAPL', 'AMZN', 'GOOGL', 'MSFT']

# Create subplots
fig, axes = plt.subplots(2, 2, figsize=(12, 12))

# Plot pie charts for each stock
for i, ticker in enumerate(tickers):
    trend_counts = stocks_data[('Trend', ticker)].value_counts()
    axes[i // 2, i % 2].pie(trend_counts, labels=trend_counts.index, autopct='%1.1f%%', startangle=90, colors=sns.color_palette('viridis'))
    axes[i // 2, i % 2].set_title(f'Trend Frequency - {ticker}')

plt.show()
